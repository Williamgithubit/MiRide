# ============================================
# Environment Variables Example for MiRide Server
# ============================================
# Copy this file to .env and fill in your values
# NEVER commit .env to version control!

# ============================================
# DATABASE CONFIGURATION
# ============================================
DB_HOST=your-database-host
DB_PORT=3306
DB_NAME=your-database-name
DB_USER=your-database-user
DB_PASSWORD=your-database-password
DB_DIALECT=mysql
DB_SSL=true

# ============================================
# SECURITY - CRITICAL
# ============================================
# JWT Secret - REQUIRED
# Generate with: node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"
# WARNING: Server will not start in production without this!
JWT_SECRET=your-jwt-secret-key-minimum-32-characters

# Admin Setup Secret (for secure admin creation endpoint)
# Generate with: node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"
ADMIN_SETUP_SECRET=your-super-secret-key-here-change-this

# ============================================
# CLIENT CONFIGURATION
# ============================================
# Client URL (for CORS and email links)
CLIENT_URL=https://your-frontend-domain.com

# ============================================
# STRIPE CONFIGURATION - REQUIRED FOR PAYMENTS
# ============================================
# Get these from https://dashboard.stripe.com/apikeys
STRIPE_SECRET_KEY=sk_live_your-stripe-secret-key
STRIPE_PUBLISHABLE_KEY=pk_live_your-stripe-publishable-key

# Webhook Secret - REQUIRED FOR PRODUCTION
# Get this from https://dashboard.stripe.com/webhooks
# Click on your webhook endpoint -> Signing secret
# WARNING: Webhooks will fail without this in production!
STRIPE_WEBHOOK_SECRET=whsec_your-webhook-signing-secret

# ============================================
# EMAIL CONFIGURATION (for notifications)
# ============================================
EMAIL_HOST=smtp.gmail.com
EMAIL_PORT=587
EMAIL_SECURE=false
EMAIL_USER=your-email@gmail.com
EMAIL_PASS=your-app-specific-password
EMAIL_FROM="MiRide <noreply@miride.com>"

# ============================================
# CLOUDINARY (for image uploads)
# ============================================
CLOUDINARY_CLOUD_NAME=your-cloud-name
CLOUDINARY_API_KEY=your-api-key
CLOUDINARY_API_SECRET=your-api-secret

# ============================================
# SERVER CONFIGURATION
# ============================================
PORT=3000
NODE_ENV=production
